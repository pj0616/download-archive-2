@Entity

    private String employeeName;

    private List<ProjectEmployeePair> projectEmployeePairs = new ArrayList<>();

    @OneToMany(mappedBy = "employee")
    private List<EmployeeSubscriptionPair> employeeSubscriptionPairs = new ArrayList<>();


    public int getID()
    {
        return ID;
    }

    public void setID(int ID)
    {
        this.ID = ID;
    }

    public String getEmployeeCode()
    {
        return employeeCode;
    }

    public void setEmployeeCode(String employeeCode)
    {
        this.employeeCode = employeeCode;
    }

    public String getEmployeeName()
    {
        return employeeName;
    }

    public void setEmployeeName(String employeeName)
    {
        this.employeeName = employeeName;
    }

    public List<ProjectEmployeePair> getProjectEmployeePairs()
    {
        return projectEmployeePairs;
    }

    public void setProjectEmployeePairs(List<ProjectEmployeePair> projectEmployeePairs)
    {
        this.projectEmployeePairs = projectEmployeePairs;
    }

    public List<EmployeeSubscriptionPair> getEmployeeSubscriptionPairs()
    {
        return employeeSubscriptionPairs;
    }

    public void setEmployeeSubscriptionPairs(List<EmployeeSubscriptionPair> employeeSubscriptionPairs)
    {
        this.employeeSubscriptionPairs = employeeSubscriptionPairs;
    }

    @Override
    public boolean equals(Object o)
    {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        if (!super.equals(o)) return false;
        Employee employee = (Employee) o;
        return ID == employee.ID &&
                Objects.equals(employeeCode, employee.employeeCode) &&
                Objects.equals(employeeName, employee.employeeName);
    }

    @Override
    public int hashCode()
    {
        return Objects.hash(super.hashCode(), ID, employeeCode, employeeName);
    }
}

@RestController
@RepositoryRestController
public class EmployeeController
{
    @Autowired
    private EmployeeRepository employeeRepository;
    @Autowired
    private EmployeeSubscriptionMapRepository employeeSubscriptionMapRepository;
    @Autowired
    private ProjectEmployeeMapRepository projectEmployeeMapRepository;

    @GetMapping("/employees")
    public ResponseEntity<List<Resource<Employee>>> getEmployees()
    {
        List<Employee> employees = employeeRepository.findAll();

        List<Resource<Employee>> employeeResources = new ArrayList<>();

        for(Employee employee : employees)
        {
            Resource<Employee> employeeResource = new Resource<>(employee);

            addLinks(employee, employeeResource);

            employeeResources.add(employeeResource);
        }

        return ResponseEntity.ok(employeeResources);
    }

    @GetMapping("/employees/{id}")
    public ResponseEntity<Resource<Employee>> getEmployees(@PathVariable("id") int id)
    {
        Optional<Employee> employeeOptional = employeeRepository.findById(id);
        if(!employeeOptional.isPresent())
        {
            throw new NotFoundException("employee-" + id);
        }
        Employee employee = employeeOptional.get();

        Resource<Employee> employeeResource = new Resource<>(employee);
        addLinks(employee, employeeResource);
        employeeResource.add(linkTo(methodOn(getClass()).getEmployees()).withRel("all"));

        return ResponseEntity.ok(employeeResource);
    }

    @PostMapping("/employees")
    public ResponseEntity<?> createEmployee(@RequestBody Employee employee)
    {
        employeeRepository.save(employee);

        URI location = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .path("/{id}")
                .buildAndExpand(employee.getID()).toUri();

        return ResponseEntity.created(location).build();
    }

    @PutMapping("/employees/{id}")
    public ResponseEntity<?> updateEmployee(@PathVariable("id") int id, @RequestBody Employee employee)
    {
        Optional<Employee> existingEmployeeOptional = employeeRepository.findById(id);
        if(!existingEmployeeOptional.isPresent())
        {
            throw new NotFoundException("employee-" + id);
        }

        Employee existingEmployee = existingEmployeeOptional.get();

        if(StringUtils.hasText(employee.getEmployeeCode()))
        {
            existingEmployee.setEmployeeCode(employee.getEmployeeCode());
        }
        if(StringUtils.hasText(employee.getEmployeeName()))
        {
            existingEmployee.setEmployeeName(employee.getEmployeeName());
        }

        employeeRepository.save(existingEmployee);


        URI location = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .buildAndExpand(existingEmployee).toUri();

        return ResponseEntity.created(location).build();
    }

    @DeleteMapping("/employees/{id}")
    public ResponseEntity<?> deleteEmployee(@PathVariable("id") int id)
    {
        List<ProjectEmployeePair> projectEmployeePairs = projectEmployeeMapRepository.findByEmployeeId(id);
        if(!projectEmployeePairs.isEmpty())
        {
            throw new ProjectViolationException("projects exist for employee-" + id);
        }

        List<EmployeeSubscriptionPair> employeeSubscriptionPairs = employeeSubscriptionMapRepository.findByEmployeeId(id);
        if(!employeeSubscriptionPairs.isEmpty())
        {
            throw new SubscriptionViolationException("subscription exist for employee-" + id);
        }

        Optional<Employee> existingEmployeeOptional = employeeRepository.findById(id);
        if(!existingEmployeeOptional.isPresent())
        {
            throw new NotFoundException("employee-" + id);
        }

        Employee employee = existingEmployeeOptional.get();
        employeeRepository.delete(employee);

        URI location = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .buildAndExpand().toUri();

        return ResponseEntity.created(location).build();
    }

    private void addLinks(Employee employee, Resource<Employee> employeeResource)
    {
        employeeResource.add(linkTo(methodOn(getClass()).getEmployees(employee.getID())).withSelfRel());

        List<Subscription> subscriptions = employeeSubscriptionMapRepository.findByEmployeeId(employee.getID())
                .stream()
                .map(EmployeeSubscriptionPair::getSubscription)
                .collect(Collectors.toList());

        for(Subscription subscription : subscriptions)
        {
            Link subscriptionLink = linkTo(methodOn(SubscriptionController.class)
                    .getSubscriptions(subscription.getID()))
                    .withRel("subscription")
                    .withTitle(subscription.getSubscriptionCode());
            employeeResource.add(subscriptionLink);
        }

        List<Project> projects = projectEmployeeMapRepository.findByEmployeeId(employee.getID())
                .stream()
                .map(ProjectEmployeePair::getProject)
                .collect(Collectors.toList());

        for(Project project : projects)
        {
            Link projectLink = linkTo(methodOn(ProjectController.class)
                    .getProjects(project.getID()))
                    .withRel("project")
                    .withTitle(project.getProjectName());
            employeeResource.add(projectLink);
        }
    }
}

@RestController
@RepositoryRestController
public class EmployeeSubscriptionMapController
{
    @Autowired
    private EmployeeSubscriptionMapRepository employeeSubscriptionMapRepository;
    @Autowired
    private EmployeeRepository employeeRepository;
    @Autowired
    private SubscriptionRepository subscriptionRepository;

    @GetMapping("/employees/subscriptions")
    public ResponseEntity<Resources<Resource<Employee>>> getEmployeesSubscriptions()
    {
        List<EmployeeSubscriptionPair> employeeSubscriptionPairs = employeeSubscriptionMapRepository.findAll();
        Map<Employee, List<Subscription>> groupedMap = employeeSubscriptionPairs.stream()
                .collect(Collectors
                        .groupingBy(EmployeeSubscriptionPair::getEmployee,
                                Collectors.mapping(EmployeeSubscriptionPair::getSubscription, Collectors.toList())));

        List<Resource<Employee>> employeeResources = new ArrayList<>();

        for(Map.Entry<Employee,List<Subscription>> e : groupedMap.entrySet())
        {
            Employee employee = e.getKey();
            List<Subscription> subscriptions = e.getValue();

            Resource<Employee> employeeResource = new Resource<>(employee);
            addLinks(employee, employeeResource, subscriptions);

            employeeResources.add(employeeResource);
        }

        Resources<Resource<Employee>> resources = new Resources<>(employeeResources);

        return ResponseEntity.ok(resources);
    }

    @GetMapping("/employees/{id}/subscriptions")
    public ResponseEntity<Resource<Employee>> getEmployeesSubscriptions(@PathVariable("id") int id)
    {
        List<EmployeeSubscriptionPair> employeeSubscriptionPairs = employeeSubscriptionMapRepository.findByEmployeeId(id);
        if(employeeSubscriptionPairs.isEmpty())
        {
            throw new NotFoundException("employee-" + id);
        }

        Optional<Employee> employeeOptional = employeeSubscriptionPairs.stream().map(EmployeeSubscriptionPair::getEmployee).findFirst();
        List<Subscription> subscriptions = employeeSubscriptionPairs
                .stream()
                .map(EmployeeSubscriptionPair::getSubscription)
                .collect(Collectors.toList());

        Employee employee = employeeOptional.get();

        Resource<Employee> employeeResource = new Resource<>(employee);
        addLinks(employee, employeeResource, subscriptions);
        employeeResource.add(linkTo(methodOn(getClass()).getEmployeesSubscriptions()).withRel("all"));
        return ResponseEntity.ok(employeeResource);
    }

    @PostMapping("/employees/{employeeId}/subscriptions/{subscriptionId}")
    public ResponseEntity<?> addSubscriptionToEmployee(@PathVariable("employeeId") int employeeId,
                                                        @PathVariable("subscriptionId") int subscriptionId)
    {
        return createAssociation(employeeId, subscriptionId);
    }

    @DeleteMapping("/employees/{employeeId}/subscriptions/{subscriptionId}")
    public ResponseEntity<?> deleteSubscriptionToEmployee(@PathVariable("employeeId") int employeeId,
                                                          @PathVariable("subscriptionId") int subscriptionId)
    {
        return deleteAssociation(employeeId, subscriptionId);
    }



    @GetMapping("/subscriptions/employees")
    public ResponseEntity<Resources<Resource<Subscription>>> getSubscriptionsEmployees()
    {
        List<EmployeeSubscriptionPair> employeeSubscriptionPairs = employeeSubscriptionMapRepository.findAll();
        Map<Subscription, List<Employee>> groupedMap = employeeSubscriptionPairs.stream()
                .collect(Collectors
                        .groupingBy(EmployeeSubscriptionPair::getSubscription,
                                Collectors.mapping(EmployeeSubscriptionPair::getEmployee, Collectors.toList())));

        List<Resource<Subscription>> subscriptionResources = new ArrayList<>();

        for(Map.Entry<Subscription,List<Employee>> e : groupedMap.entrySet())
        {
            Subscription subscription = e.getKey();
            List<Employee> employees = e.getValue();

            Resource<Subscription> subscriptionResource = new Resource<>(subscription);
            addLinks(subscription, subscriptionResource, employees);

            subscriptionResources.add(subscriptionResource);
        }

        Resources<Resource<Subscription>> resources = new Resources<>(subscriptionResources);

        return ResponseEntity.ok(resources);
    }

    @GetMapping("/subscriptions/{id}/employees")
    public ResponseEntity<Resource<Subscription>> getSubscriptionsEmployees(@PathVariable("id") int id)
    {
        List<EmployeeSubscriptionPair> employeeSubscriptionPairs = employeeSubscriptionMapRepository.findBySubscriptionId(id);
        if(employeeSubscriptionPairs.isEmpty())
        {
            throw new NotFoundException("subscription-" + id);
        }

        Optional<Subscription> subscriptionOptional = employeeSubscriptionPairs.stream().map(EmployeeSubscriptionPair::getSubscription).findFirst();
        List<Employee> employees = employeeSubscriptionPairs
                .stream()
                .map(EmployeeSubscriptionPair::getEmployee)
                .collect(Collectors.toList());

        Subscription subscription = subscriptionOptional.get();

        Resource<Subscription> subscriptionResource = new Resource<>(subscription);
        addLinks(subscription, subscriptionResource, employees);
        subscriptionResource.add(linkTo(methodOn(getClass()).getSubscriptionsEmployees()).withRel("all"));
        return ResponseEntity.ok(subscriptionResource);
    }

    @PostMapping("/subscriptions/{subscriptionId}/employees/{employeeId}")
    public ResponseEntity<?> addEmployeeToSubscription(@PathVariable("subscriptionId") int subscriptionId,
                                                       @PathVariable("employeeId") int employeeId)
    {
        return createAssociation(employeeId, subscriptionId);
    }

    @DeleteMapping("/subscriptions/{subscriptionId}/employees/{employeeId}")
    public ResponseEntity<?> deleteEmployeeFromSubscription(@PathVariable("subscriptionId") int subscriptionId,
                                                            @PathVariable("employeeId") int employeeId)
    {
        return deleteAssociation(employeeId, subscriptionId);
    }

    private ResponseEntity<?> createAssociation(int employeeId, int subscriptionId)
    {
        Optional<Employee> employeeOptional = employeeRepository.findById(employeeId);
        if (!employeeOptional.isPresent())
        {
            throw new NotFoundException("employee-" + employeeId);
        }
        Optional<Subscription> subscriptionOptional = subscriptionRepository.findById(subscriptionId);
        if (!subscriptionOptional.isPresent())
        {
            throw new NotFoundException("subscription-" + employeeId);
        }

        Employee employee = employeeOptional.get();
        Subscription subscription = subscriptionOptional.get();

        //  Check to make sure this association does not already exists
        List<Subscription> subscriptions = employee.getEmployeeSubscriptionPairs().stream().map(EmployeeSubscriptionPair::getSubscription).collect(Collectors.toList());
        if(!subscriptions.contains(subscription))
        {
            EmployeeSubscriptionPair employeeSubscriptionPair = new EmployeeSubscriptionPair(employee, subscription);
            employeeSubscriptionMapRepository.save(employeeSubscriptionPair);

            URI location = ServletUriComponentsBuilder
                    .fromCurrentRequest()
                    .buildAndExpand(employee.getID(), subscription.getID())
                    .toUri();
            return ResponseEntity.created(location).build();
        }

        throw new AlreadyExistsException("employeeId-" + employeeId + ", subscriptionId-" + subscriptionId);
    }

    private ResponseEntity<?> deleteAssociation(int employeeId, int subscriptionId)
    {
        Optional<EmployeeSubscriptionPair> employeeSubscriptionPairOptional =
                employeeSubscriptionMapRepository.findByEmployeeIdSubscriptionId(employeeId, subscriptionId);
        if(!employeeSubscriptionPairOptional.isPresent())
        {
            throw new NotFoundException("employeeId-" + employeeId + ", subscriptionId-" + subscriptionId);
        }

        EmployeeSubscriptionPair employeeSubscriptionPair = employeeSubscriptionPairOptional.get();
        employeeSubscriptionMapRepository.delete(employeeSubscriptionPair);

        URI location = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .buildAndExpand(employeeSubscriptionPair).toUri();

        return ResponseEntity.created(location).build();
    }

    private void addLinks(Employee employee, Resource<Employee> employeeResource, List<Subscription> subscriptions)
    {
        employeeResource.add(linkTo(methodOn(getClass()).getEmployeesSubscriptions(employee.getID())).withSelfRel());

        for(Subscription subscription : subscriptions)
        {
            Link subscriptionLink = linkTo(methodOn(SubscriptionController.class)
                    .getSubscriptions(subscription.getID()))
                    .withRel("subscription")
                    .withTitle(subscription.getSubscriptionName());
            employeeResource.add(subscriptionLink);
        }
    }

    private void addLinks(Subscription subscription, Resource<Subscription> subscriptionResource, List<Employee> employees)
    {
        subscriptionResource.add(linkTo(methodOn(getClass()).getSubscriptionsEmployees(subscription.getID())).withSelfRel());

        for(Employee employee : employees)
        {
            Link employeeLink = linkTo(methodOn(EmployeeController.class)
                    .getEmployees(employee.getID()))
                    .withRel("employee")
                    .withTitle(employee.getEmployeeName());
            subscriptionResource.add(employeeLink);
        }
    }
}

    private ResponseEntity<?> createAssociation(int projectId, int employeeId)
    {
        Optional<Project> projectOptional = projectRepository.findById(projectId);
        if (!projectOptional.isPresent())
        {
            throw new NotFoundException("project-" + projectId);
        }
        Optional<Employee> employeeOptional = employeeRepository.findById(employeeId);
        if (!employeeOptional.isPresent())
        {
            throw new NotFoundException("employee-" + employeeId);
        }

        Project project = projectOptional.get();
        Employee employee = employeeOptional.get();

        //  Check to make sure this association does not already exists
        List<Employee> employees = project.getProjectEmployeePairs().stream().map(ProjectEmployeePair::getEmployee).collect(Collectors.toList());
        if(!employees.contains(employee))
        {
            ProjectEmployeePair projectEmployeePair = new ProjectEmployeePair(project, employee);
            projectEmployeeMapRepository.save(projectEmployeePair);

            URI location = ServletUriComponentsBuilder
                    .fromCurrentRequest()
                    .buildAndExpand(project.getID(), employee.getID())
                    .toUri();
            return ResponseEntity.created(location).build();
        }

        throw new AlreadyExistsException("projectId-" + projectId + ", employeeId-" + employeeId);
    }

    private ResponseEntity<?> deleteAssociation(int projectId, int employeeId)
    {
        Optional<ProjectEmployeePair> projectEmployeePairOptional =
                projectEmployeeMapRepository.findByProjectIdEmployeeId(projectId, employeeId);
        if(!projectEmployeePairOptional.isPresent())
        {
            throw new NotFoundException("projectId-" + projectId + ", employeeId-" + employeeId);
        }

        ProjectEmployeePair projectEmployeePair = projectEmployeePairOptional.get();
        projectEmployeeMapRepository.delete(projectEmployeePair);

        URI location = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .buildAndExpand(projectEmployeePair).toUri();

        return ResponseEntity.created(location).build();
    }

    private void addLinks(Project project, Resource<Project> projectResource, List<Employee> employees)
    {
        projectResource.add(linkTo(methodOn(getClass()).getProjectsEmployees(project.getID())).withSelfRel());

        for(Employee employee : employees)
        {
            Link employeeLink = linkTo(methodOn(EmployeeController.class)
                    .getEmployees(employee.getID()))
                    .withRel("employee")
                    .withTitle(employee.getEmployeeName());
            projectResource.add(employeeLink);
        }
    }

    private void addLinks(Employee employee, Resource<Employee> employeeResource, List<Project> projects)
    {
        employeeResource.add(linkTo(methodOn(getClass()).getEmployeesProjects(employee.getID())).withSelfRel());

        for(Project project : projects)
        {
            Link projectLink = linkTo(methodOn(ProjectController.class)
                    .getProjects(project.getID()))
                    .withRel("project")
                    .withTitle(project.getProjectName());
            employeeResource.add(projectLink);
        }
    }
