DevOps : Add JQL to Repository classes

Created by Nicholas Guner on Apr 27, 2020

We provide functionality to Pair controllers to get all employees for the specified project, all project for specified employee, all subscriptions for specified employee, etc.

Changes to Repository classes

public interface EmployeeSubscriptionMapRepository extends JpaRepository<EmployeeSubscriptionPair, Integer>
{
    @Query("select p from EmployeeSubscriptionPair p where EMPLOYEE_ID = :id")
    List<EmployeeSubscriptionPair> findByEmployeeId(@Param("id") int id);

    @Query("select p from EmployeeSubscriptionPair p where p.subscription.ID = :id")
    List<EmployeeSubscriptionPair> findBySubscriptionId(@Param("id") int id);
}


public interface ProjectEmployeeMapRepository extends JpaRepository<ProjectEmployeePair, Integer>
{
    @Query("select p from ProjectEmployeePair p where PROJECT_ID = :id")
    List<ProjectEmployeePair> findByProjectId(@Param("id") int id);

    @Query("select p from ProjectEmployeePair p where p.employee.ID = :id")
    List<ProjectEmployeePair> findByEmployeeId(@Param("id") int id);
}

Changes to the Controller Map Classes

@RestController
@RepositoryRestController
public class ProjectEmployeeMapController
{
    @Autowired
    private ProjectEmployeeMapRepository projectEmployeeMapRepository;

    @GetMapping("/projects/employees")
    public ResponseEntity<List<Project>> getProjectsEmployees()
    {
        List<ProjectEmployeePair> projectEmployeePairs = projectEmployeeMapRepository.findAll();
        Map<Project, List<Employee>> groupedMap = projectEmployeePairs.stream()
                .collect(Collectors
                        .groupingBy(ProjectEmployeePair::getProject,
                                Collectors.mapping(ProjectEmployeePair::getEmployee, Collectors.toList())));

        List<Project> result = new ArrayList<>();
        for(Map.Entry<Project,List<Employee>> e : groupedMap.entrySet())
        {
            result.add(e.getKey());
        }

        return ResponseEntity.ok(result);
    }

    @GetMapping("/projects/{id}/employees")
    public ResponseEntity<List<Employee>> getProjectsEmployees(@PathVariable("id") int id)
