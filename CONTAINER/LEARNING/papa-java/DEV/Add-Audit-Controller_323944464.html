Create new controller under the new controller/audit package

@RequestMapping(value = "/audit")
@RestController
public class SubscriptionAuditController
{
    @Autowired
    private SubscriptionRepository subscriptionRepository;
    @Autowired
    private EmployeeSubscriptionMapRepository employeeSubscriptionMapRepository;

    @GetMapping("/subscriptions")
    public ResponseEntity<List<Resource<Subscription>>> getSubscriptions()
    {
        List<Subscription> subscriptions = subscriptionRepository.findAllWithAudit();

        List<Resource<Subscription>> subscriptionResources = new ArrayList<>();

        for(Subscription subscription : subscriptions)
        {
            Resource<Subscription> subscriptionResource = new Resource<>(subscription);

            addLinks(subscription, subscriptionResource);

            subscriptionResources.add(subscriptionResource);
        }

        return ResponseEntity.ok(subscriptionResources);
    }

    @GetMapping("/subscriptions/{id}")
    public ResponseEntity<List<Resource<Subscription>>> getSubscriptions(@PathVariable("id") int id)
    {
        Optional<Subscription> subscriptionOptional = subscriptionRepository.findById(id);
        if(!subscriptionOptional.isPresent())
        {
            throw new NotFoundException("subscription-" + id);
        }
        Subscription liveSubscription = subscriptionOptional.get();

        List<Subscription> subscriptions = subscriptionRepository.findByAuditId(liveSubscription.getAuditId());

        List<Resource<Subscription>> subscriptionResources = new ArrayList<>();

        for(Subscription subscription : subscriptions)
        {
            Resource<Subscription> subscriptionResource = new Resource<>(subscription);

            addLinks(subscription, subscriptionResource);

            subscriptionResources.add(subscriptionResource);
        }

        return ResponseEntity.ok(subscriptionResources);
    }

    private void addLinks(Subscription subscription, Resource<Subscription> subscriptionResource)
    {
        subscriptionResource.add(linkTo(methodOn(getClass()).getSubscriptions(subscription.getID())).withSelfRel());

        List<Employee> employees = employeeSubscriptionMapRepository.findByEmployeeId(subscription.getID())
                .stream()
                .map(EmployeeSubscriptionPair::getEmployee)
                .collect(Collectors.toList());

        for(Employee employee : employees)
        {
            Link employeeLink = linkTo(methodOn(EmployeeController.class)
                    .getEmployees(employee.getID()))
                    .withRel("employee")
                    .withTitle(employee.getEmployeeName());
            subscriptionResource.add(employeeLink);
        }
    }
}

@GetMapping("/subscriptions")
public ResponseEntity<List<Resource<Subscription>>> getSubscriptions(@RequestParam(defaultValue = "false") boolean showAuditOnly)
{
    List<Subscription> subscriptions;
    if(showAuditOnly)
    {
        subscriptions = subscriptionRepository.findAllAudit();
    }
    else
    {
        subscriptions = subscriptionRepository.findAllWithAudit();
    }
    List<Resource<Subscription>> subscriptionResources = new ArrayList<>();
    for(Subscription subscription : subscriptions)
    {
        Resource<Subscription> subscriptionResource = new Resource<>(subscription);
        addLinks(subscription, subscriptionResource);
        subscriptionResources.add(subscriptionResource);
    }
    return ResponseEntity.ok(subscriptionResources);
}

Refactoring

One on big problems with current code is the fact that addLinks method appears in both regular and audit controller. If we ever need to make a change in this method, we must remember to make change in both places.

Create new class in contoller/links package

@Component
public class ProjectEmployeeLinks
{
    public List<Resource<Project>> createProjectEmployeeResources(Map<Project, List<Employee>> projectEmployeesMap,
                                                                  ProjectEmployeeBaseController baseController)
    {
        List<Resource<Project>> projectResources = new ArrayList<>();

        for(Map.Entry<Project,List<Employee>> e : projectEmployeesMap.entrySet())
        {
            Project project = e.getKey();
            List<Employee> employees = e.getValue();

            Resource<Project> projectResource = baseController.createProjectEmployeeResource(project, employees);

            projectResources.add(projectResource);
        }

        return projectResources;
    }

    public List<Resource<Employee>> createEmployeeProjectResources(Map<Employee, List<Project>> employeeProjectsMap,
                                                                    ProjectEmployeeBaseController baseController)
    {
        List<Resource<Employee>> employeeResources = new ArrayList<>();

        for(Map.Entry<Employee, List<Project>> e : employeeProjectsMap.entrySet())
        {
            Employee employee = e.getKey();
            List<Project> projects = e.getValue();

            Resource<Employee> employeeResource = baseController.createEmployeeProjectResource(employee, projects);

            employeeResources.add(employeeResource);
        }

        return employeeResources;
    }


    public void addLinks(Project project,
                         Resource<Project> projectResource,
                         List<Employee> employees,
                         Link link)
    {
        projectResource.add(link);

        for(Employee employee : employees)
        {
            Link employeeLink = linkTo(methodOn(EmployeeController.class)
                    .getEmployees(employee.getID()))
                    .withRel("employee")
                    .withTitle(employee.getEmployeeName());
            projectResource.add(employeeLink);
        }
    }

    public void addLinks(Employee employee, Resource<Employee> employeeResource, List<Project> projects, Link link)
    {
        employeeResource.add(link);

        for(Project project : projects)
        {
            Link projectLink = linkTo(methodOn(ProjectController.class)
                    .getProjects(project.getID()))
                    .withRel("project")
                    .withTitle(project.getProjectName());
            employeeResource.add(projectLink);
        }
    }
}

public abstract class ProjectEmployeeBaseController
{
    abstract public Resource<Project> createProjectEmployeeResource(Project project, List<Employee> employees);
    abstract public Resource<Employee> createEmployeeProjectResource(Employee employee, List<Project> projects);
}

@RestController
public class ProjectEmployeeController extends ProjectEmployeeBaseController
{
    @Autowired
    private ProjectEmployeeMapRepository projectEmployeeMapRepository;
    @Autowired
    private ProjectRepository projectRepository;
    @Autowired
    private EmployeeRepository employeeRepository;
    @Autowired
    private ProjectEmployeeLinks projectEmployeeLinks;


    @GetMapping("/projects/employees")
    public ResponseEntity<Resources<Resource<Project>>> getProjectsEmployees()
    {
        List<ProjectEmployeePair> projectEmployeePairs = projectEmployeeMapRepository.findAll();
        Map<Project, List<Employee>> groupedMap = projectEmployeePairs.stream()
                .collect(Collectors
                        .groupingBy(ProjectEmployeePair::getProject,
                                Collectors.mapping(ProjectEmployeePair::getEmployee, Collectors.toList())));

        List<Resource<Project>> projectResources = projectEmployeeLinks.createProjectEmployeeResources(groupedMap, this);

        Resources<Resource<Project>> resources = new Resources<>(projectResources);

        return ResponseEntity.ok(resources);
    }

    @GetMapping("/projects/{id}/employees")
    public ResponseEntity<Resource<Project>> getProjectsEmployees(@PathVariable("id") int id)
    {
        List<ProjectEmployeePair> projectEmployeePairs = projectEmployeeMapRepository.findByProjectId(id);
        if(projectEmployeePairs.isEmpty())
        {
            throw new NotFoundException("project-" + id);
        }

        Optional<Project> projectOptional = projectEmployeePairs.stream().map(ProjectEmployeePair::getProject).findFirst();
        List<Employee> employees = projectEmployeePairs
                .stream()
                .map(ProjectEmployeePair::getEmployee)
                .collect(Collectors.toList());

        Project project = projectOptional.get();

        Resource<Project> projectResource = createProjectEmployeeResource(project, employees);
        projectResource.add(linkTo(methodOn(getClass()).getProjectsEmployees()).withRel("all"));
        return ResponseEntity.ok(projectResource);
    }

    @PostMapping("/projects/{projectId}/employees/{employeeId}")
    public ResponseEntity<?> addEmployeeToProject(@PathVariable("projectId") int projectId,
                                                  @PathVariable("employeeId") int employeeId)
    {
        return createAssociation(projectId, employeeId);
    }

    @DeleteMapping("/projects/{projectId}/employees/{employeeId}")
    public ResponseEntity<?> deleteEmployeeFromProject(@PathVariable("projectId") int projectId,
                                                        @PathVariable("employeeId") int employeeId)
    {
        return deleteAssociation(projectId, employeeId);
    }


    @GetMapping("/employees/projects")
    public ResponseEntity<Resources<Resource<Employee>>> getEmployeesProjects()
    {
        List<ProjectEmployeePair> projectEmployeePairs = projectEmployeeMapRepository.findAll();
        Map<Employee, List<Project>> groupedMap = projectEmployeePairs.stream()
                .collect(Collectors
                        .groupingBy(ProjectEmployeePair::getEmployee,
                                Collectors.mapping(ProjectEmployeePair::getProject, Collectors.toList())));

        List<Resource<Employee>> employeeResources = projectEmployeeLinks.createEmployeeProjectResources(groupedMap, this);

        Resources<Resource<Employee>> resources = new Resources<>(employeeResources);

        return ResponseEntity.ok(resources);
    }

    @GetMapping("/employees/{id}/projects")
    public ResponseEntity<Resource<Employee>> getEmployeesProjects(@PathVariable("id") int id)
    {
        List<ProjectEmployeePair> projectEmployeePairs = projectEmployeeMapRepository.findByEmployeeId(id);
        if(projectEmployeePairs.isEmpty())
        {
            throw new NotFoundException("employee-" + id);
        }

        Optional<Employee> employeeOptional = projectEmployeePairs.stream().map(ProjectEmployeePair::getEmployee).findFirst();
        List<Project> projects = projectEmployeePairs
                .stream()
                .map(ProjectEmployeePair::getProject)
                .collect(Collectors.toList());

        Employee employee = employeeOptional.get();

        Resource<Employee> employeeResource = createEmployeeProjectResource(employee, projects);
        employeeResource.add(linkTo(methodOn(getClass()).getEmployeesProjects()).withRel("all"));
        return ResponseEntity.ok(employeeResource);
    }

    @PostMapping("/employees/{employeeId}/projects/{projectId}")
    public ResponseEntity<?> addProjectToEmployee(@PathVariable("employeeId") int employeeId,
                                                  @PathVariable("projectId") int projectId)
    {
        return createAssociation(projectId, employeeId);
    }

    @DeleteMapping("/employees/{employeeId}/projects/{projectId}")
    public ResponseEntity<?> deleteProjectFromEmployee(@PathVariable("employeeId") int employeeId,
                                                       @PathVariable("projectId") int projectId)
    {
        return deleteAssociation(projectId, employeeId);
    }


    private ResponseEntity<?> createAssociation(int projectId, int employeeId)
    {
        Optional<Project> projectOptional = projectRepository.findById(projectId);
        if (!projectOptional.isPresent())
        {
            throw new NotFoundException("project-" + projectId);
        }
        Optional<Employee> employeeOptional = employeeRepository.findById(employeeId);
        if (!employeeOptional.isPresent())
        {
            throw new NotFoundException("employee-" + employeeId);
        }

        Project project = projectOptional.get();
        Employee employee = employeeOptional.get();

        //  Check to make sure this association does not already exists
        List<Employee> employees = project.getProjectEmployeePairs().stream().map(ProjectEmployeePair::getEmployee).collect(Collectors.toList());
        if(!employees.contains(employee))
        {
            ProjectEmployeePair projectEmployeePair = new ProjectEmployeePair(project, employee);
            projectEmployeeMapRepository.save(projectEmployeePair);

            URI location = ServletUriComponentsBuilder
                    .fromCurrentRequest()
                    .buildAndExpand(project.getID(), employee.getID())
                    .toUri();
            return ResponseEntity.created(location).build();
        }

        throw new AlreadyExistsException("projectId-" + projectId + ", employeeId-" + employeeId);
    }

    private ResponseEntity<?> deleteAssociation(int projectId, int employeeId)
    {
        Optional<ProjectEmployeePair> projectEmployeePairOptional =
                projectEmployeeMapRepository.findByProjectIdEmployeeId(projectId, employeeId);
        if(!projectEmployeePairOptional.isPresent())
        {
            throw new NotFoundException("projectId-" + projectId + ", employeeId-" + employeeId);
        }

        ProjectEmployeePair projectEmployeePair = projectEmployeePairOptional.get();
        projectEmployeePair.setOutUtc(ZonedDateTime.now());
        projectEmployeeMapRepository.save(projectEmployeePair);

        URI location = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .buildAndExpand(projectEmployeePair).toUri();

        return ResponseEntity.created(location).build();
    }


    public Resource<Project> createProjectEmployeeResource(Project project, List<Employee> employees)
    {
        Resource<Project> projectResource = new Resource<>(project);
        Link link = linkTo(methodOn(getClass()).getProjectsEmployees(project.getID())).withSelfRel();
        projectEmployeeLinks.addLinks(project, projectResource, employees, link);
        return projectResource;
    }

    public Resource<Employee> createEmployeeProjectResource(Employee employee, List<Project> projects)
    {
        Resource<Employee> employeeResource = new Resource<>(employee);
        Link link = linkTo(methodOn(getClass()).getEmployeesProjects(employee.getID())).withSelfRel();
        projectEmployeeLinks.addLinks(employee, employeeResource, projects, link);
        return employeeResource;
    }

}
