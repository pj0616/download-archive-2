@GetMapping("/employees/{id}/subscriptions")
public ResponseEntity<Resource<Employee>> getEmployeesSubscriptions(@PathVariable("id") int id)
{
    List<EmployeeSubscriptionPair> employeeSubscriptionPairs = employeeSubscriptionMapRepository.findByEmployeeId(id);
    if(employeeSubscriptionPairs.isEmpty())
    {
        throw new NotFoundException("employee-" + id);
    }
    Optional<Employee> employeeOptional = employeeSubscriptionPairs.stream().map(EmployeeSubscriptionPair::getEmployee).findFirst();
    List<Subscription> subscriptions = employeeSubscriptionPairs
            .stream()
            .map(EmployeeSubscriptionPair::getSubscription)
            .collect(Collectors.toList());
    Employee employee = employeeOptional.get();
    Resource<Employee> employeeResource = createEmployeeSubscriptionResource(employee, subscriptions);
    employeeResource.add(linkTo(methodOn(getClass()).getEmployeesSubscriptions()).withRel("all"));
    return ResponseEntity.ok(employeeResource);
}
