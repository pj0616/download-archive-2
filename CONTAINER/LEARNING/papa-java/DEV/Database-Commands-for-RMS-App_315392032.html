DevOps : Database Commands for RMS App

Created by Nicholas Guner on Apr 27, 2020

Following will need to be executed from the promproxy account:

Set active schema;

Command:

set schema SOD_TEST


Refer to create.sql for DDL commands to create tables and indices

create table employee (id integer not null, created_by varchar(255), in_utc timestamp not null, last_modified_by varchar(255), out_utc timestamp, audit_id varchar(255), department varchar(255), employee_number varchar(255), guid varchar(255), kerberos varchar(255), location varchar(255), name varchar(255), phone varchar(255), title varchar(255), team_id integer not null, primary key (id));
create table rms_node (id integer not null, created_by varchar(255), in_utc timestamp not null, last_modified_by varchar(255), out_utc timestamp, audit_id varchar(255), code varchar(255), name varchar(255), node_type varchar(255), team_id integer, primary key (id));
create table rms_node_map (id integer not null, created_by varchar(255), in_utc timestamp not null, last_modified_by varchar(255), out_utc timestamp, audit_id varchar(255), child_id integer, parent_id integer, primary key (id));
create table team (id integer not null, created_by varchar(255), in_utc timestamp not null, last_modified_by varchar(255), out_utc timestamp, audit_id varchar(255), primary key (id));
create unique index UKcbw6ome5i54khsmaeaax2y36o on employee (audit_id, out_utc);
create unique index UKqy9bi7sv21otr18x3ankrag2 on rms_node (audit_id, out_utc);
create unique index UKm1h091sd0inao94jknd11xann on rms_node_map (audit_id, out_utc);
create unique index UK1xc53ip1ma678dfmgd4kgl462 on team (audit_id, out_utc);
alter table employee add constraint FK8d7lrsr6kwirr93rx0tafnoqa foreign key (team_id) references team;

alter table rms_node_map add constraint FKnmejcil40cmp27chidn12ys70 foreign key (parent_id) references rms_node;

Create HIBERNATE_SEQUENCE

create sequence HIBERNATE_SEQUENCE;
alter sequence HIBERNATE_SEQUENCE restart with 101;
