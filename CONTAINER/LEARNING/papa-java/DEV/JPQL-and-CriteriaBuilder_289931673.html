DevOps : JPQL and CriteriaBuilder

Created by Nicholas Guner on Apr 20, 2020

In this tutorial, we expand on the functionality in Library Application Example by showing how to query data using JPQL and Criteria APIs.

We will create two new controller classes: CriterialController and PjqlController and provide REST API to get all User objects and to get one User object by id.

Using Criteria API

@RestController
@RepositoryRestController
public class CriteriaController
{
    @Autowired
    private EntityManager entityManager;

    @GetMapping("/criteria/readers")
    public @ResponseBody
    ResponseEntity<?> getAllReaders()
    {
        CriteriaBuilder cb = entityManager.getCriteriaBuilder();
        CriteriaQuery<Reader> readerQuery = cb.createQuery(Reader.class);
        Root<Reader> root = readerQuery.from(Reader.class);
        readerQuery.select(root);
        TypedQuery<Reader> typedQuery = entityManager.createQuery(readerQuery);
        List<Reader> readers = typedQuery.getResultList();

        List<Reader> linkedReaders = readers.stream()
                .map(u -> {u.add(linkTo(methodOn(getClass()).getAllReaders()).slash( "/" + u.getReaderId()).withSelfRel()); return u;})
                .collect(Collectors.toList());

        Resources<Reader> resources = new Resources<>(linkedReaders);
        resources.add(linkTo(methodOn(getClass()).getAllReaders()).withSelfRel());
        return ResponseEntity.ok(resources);
    }

    @GetMapping("/criteria/readers/{id}")
    public ResponseEntity<Resource<Reader>> getOneReader(@PathVariable("id") int id)
    {
        CriteriaBuilder cb = entityManager.getCriteriaBuilder();
        CriteriaQuery<Reader> readerQuery = cb.createQuery(Reader.class);
        Root<Reader> root = readerQuery.from(Reader.class);

        ParameterExpression<Integer> idParam = cb.parameter(Integer.class);
        readerQuery.select(root).where(cb.equal(root.get("readerId"), idParam));

        TypedQuery<Reader> typedQuery = entityManager.createQuery(readerQuery);
        typedQuery.setParameter(idParam, id);
        Reader reader = typedQuery.getSingleResult();

        Resource<Reader> resource = new Resource<>(reader);
        ControllerLinkBuilder linkTo = linkTo(methodOn(getClass()).getAllReaders());
        resource.add(linkTo.withRel("all-readers"));
        return ResponseEntity.ok(resource);
    }
}


Using JPQL

@RestController
@RepositoryRestController
public class JpqlController
{
    @Autowired
    private EntityManager entityManager;

    @GetMapping("/jpql/readers")
    public @ResponseBody
    ResponseEntity<?> getAllReaders()
    {
        List<Reader> readers = entityManager.createQuery("SELECT u from Reader u", Reader.class).getResultList();

        List<Reader> linkedReaders = readers.stream()
                .map(u -> {u.add(linkTo(methodOn(getClass()).getAllReaders()).slash( "/" + u.getReaderId()).withSelfRel()); return u;})
                .collect(Collectors.toList());

        Resources<Reader> resources = new Resources<>(linkedReaders);
        resources.add(linkTo(methodOn(getClass()).getAllReaders()).withSelfRel());
        return ResponseEntity.ok(resources);
    }

    @GetMapping("/jpql/readers/{id}")
    public ResponseEntity<Resource<Reader>> getOneReader(@PathVariable("id") int id)
    {
        TypedQuery<Reader> readerQuery = entityManager.createQuery("SELECT u from Reader u WHERE u.readerId = :id", Reader.class);
        readerQuery.setParameter("id", id);
        Reader reader = readerQuery.getSingleResult();

        Resource<Reader> resource = new Resource<>(reader);
        ControllerLinkBuilder linkTo = linkTo(methodOn(getClass()).getAllReaders());
        resource.add(linkTo.withRel("all-readers"));
        return ResponseEntity.ok(resource);
    }

    @GetMapping("/jpql/books/{id}")
    public @ResponseBody
    ResponseEntity<?> getBook(@PathVariable("id") int id)
    {
        Book book = entityManager.find(Book.class, id);
        Resource<Book> resource = new Resource<>(book);
        resource.add(linkTo(methodOn(getClass()).getBook(id)).withSelfRel());
        return ResponseEntity.ok(resource);
    }
}

Access Data from EntityManager Directly

The code to query for object by ID looks incredibly complex, especially in the Criteria API case. We can accomplish same functionality using the find method of the EntityManager:

@GetMapping("/jpql/books/{id}")
public @ResponseBody
ResponseEntity<?> getBook(@PathVariable("id") int id)
{
    Book book = entityManager.find(Book.class, id);
    Resource<Book> resource = new Resource<>(book);
    resource.add(linkTo(methodOn(getClass()).getBook(id)).withSelfRel());
    return ResponseEntity.ok(resource);
}
