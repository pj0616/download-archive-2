Simplify UserController to support GET-ALL and GET-ONE Operations Only

Before proceeding, we will simplify our UserController class since we will only be interested in GET-ALL and GET-ONE functionality in the current tutorial.

@RestController
@RepositoryRestController
public class UserController
{
    private Logger logger = LoggerFactory.getLogger(getClass());
    @Autowired
    private UserRepository userRepository;

    @GetMapping("/users")
    public @ResponseBody ResponseEntity<?> getAll()
    {
        List<User> users = userRepository.findAll().stream()
                .map(u -> {u.add(linkTo(methodOn(getClass()).getAll()).slash( "/" + u.getUserID()).withSelfRel()); return u;})
                .collect(Collectors.toList());

        Resources<User> resources = new Resources<>(users);
        resources.add(linkTo(methodOn(getClass()).getAll()).withSelfRel());
        return ResponseEntity.ok(resources);
    }

    @GetMapping("/users/{id}")
    public @ResponseBody
    User getOne(@PathVariable("id") long id)
    {
        Optional<User> currentUserOptional = userRepository.findById(id);
        return currentUserOptional.get();
    }
}

    return ResponseEntity.ok(resources);
}

Create Entity and Repository Classes for USER_ORDERS

Entity class - UserOrders

@Entity
public class UserOrders extends ResourceSupport
{
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    @Column(name="ID")
    private long userOrdersID;
    private long userID;
    private long orderID;

    public long getUserOrdersID()
    {
        return userOrdersID;
    }

    public void setUserOrdersID(long userOrdersID)
    {
        this.userOrdersID = userOrdersID;
    }

    public long getUserID()
    {
        return userID;
    }

    public void setUserID(long userID)
    {
        this.userID = userID;
    }

    public long getOrderID()
    {
        return orderID;
    }

    public void setOrderID(long orderID)
    {
        this.orderID = orderID;
    }

    @Override
    public String toString()
    {
        return "UserOrders{" +
                "userOrdersID=" + userOrdersID +
                ", userID=" + userID +
                ", orderID=" + orderID +
                '}';
    }
}

Controller class for Users-to-Orders

@RestController
@RepositoryRestController
public class UserOrdersController
{
    @Autowired
    private UserOrdersRepository userOrdersRepository;
    @Autowired
    private UserRepository userRepository;
    @Autowired
    private OrderDetailRepository orderDetailRepository;

    @GetMapping("/users/orders")
    public ResponseEntity<Resources<Resource<?>>> getAllUsersOrders()
    {
        List<Resource<?>> resourceResult = new ArrayList<>();
        Set<Long> processedUserIdSet = new HashSet<>();

        List<UserOrders> userOrdersList = userOrdersRepository.findAll();
        for(UserOrders usersOrders : userOrdersList)
        {
            long userId = usersOrders.getUserID();
            if(processedUserIdSet.add(userId))
            {
                User user = userRepository.findById(userId).get();
                Resource<User> userResource = new Resource<>(user, createUserLink(userId));
                resourceResult.add(userResource);

                List<UserOrders> ordersForUser = userOrdersRepository.findAll().stream().filter(u -> u.getUserID() == userId).collect(Collectors.toList());
                for(UserOrders orders : ordersForUser)
                {
                    long orderId = orders.getOrderID();
                    OrderDetail orderDetail = orderDetailRepository.findById(orderId).get();
                    Resource<OrderDetail> orderDetailResource = new Resource<>(orderDetail, createOrderDetailLink(orderId));
                    resourceResult.add(orderDetailResource);
                }
            }
        }

        Resources<Resource<?>> resources = new Resources<>(resourceResult, linkTo(methodOn(getClass()).getAllUsersOrders()).withSelfRel());
        return ResponseEntity.ok(resources);
    }

    private Link createUserLink(long userId)
    {
        return linkTo(methodOn(UserController.class).getOne(userId)).withRel("user");
    }

    private Link createOrderDetailLink(long orderId)
    {
        return linkTo(methodOn(OrderDetailController.class).getOne(orderId)).withRel("order");
    }
}

@GetMapping("/users/{userId}/orders")
ResponseEntity<?> getAllOrdersByUser(@PathVariable("userId") long userId)
{
    List<Resource<?>> resourceResult = new ArrayList<>();
    User user = userRepository.findById(userId).get();
    Resource<User> userResource = new Resource<>(user, createUserLink(userId));
    resourceResult.add(userResource);
    List<UserOrders> userOrdersFilteredList = userOrdersRepository.findAll().stream().filter(u -> u.getUserID() == userId).collect(Collectors.toList());
    for(UserOrders userOrders : userOrdersFilteredList)
    {
        long orderId = userOrders.getOrderID();
        OrderDetail orderDetail = orderDetailRepository.findById(orderId).get();
        Resource<OrderDetail> orderDetailResource = new Resource<>(orderDetail, createOrderDetailLink(orderId));
        resourceResult.add(orderDetailResource);
    }
    Resources<Resource<?>> resources = new Resources<>(resourceResult, linkTo(methodOn(getClass()).getAllOrdersByUser(userId)).withSelfRel());
    return ResponseEntity.ok(resources);
}
